Production Database

$ psql postgres
$ postgres=# CREATE DATABASE "chitter";
$ postgres=# \c chitter;
$ chitter=# CREATE TABLE users(id SERIAL PRIMARY KEY, username VARCHAR(60) UNIQUE NOT NULL);
$ chitter=# CREATE TABLE peeps(id SERIAL PRIMARY KEY, message VARCHAR(120) NOT NULL, time TIME);
$ chitter=# ALTER TABLE peeps ADD COLUMN username VARCHAR(60) REFERENCES users(username);
